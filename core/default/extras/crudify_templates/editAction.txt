$request = new RivetyCore_Request($this->getRequest());
$ENTITY_NAME_dbtable = new MODEL_CLASS_NAME();
$ROW_VAR_NAME = $ENTITY_NAME_dbtable->emptyRowArray();
if ($request->has('ENTITY_NAME_ID_COLUMN_NAME') && !empty($request->ENTITY_NAME_ID_COLUMN_NAME))
{
	$ROW_VAR_NAME = $ENTITY_NAME_dbtable->fetchRowArray($ENTITY_NAME_dbtable->getAdapter()->quoteInto('ID_COLUMN_NAME = ?', $request->ENTITY_NAME_ID_COLUMN_NAME));
}
if ($this->getRequest()->isPost())
{
	POST_VALUE_ASSIGNMENT
	REQUEST_VALIDATION
	if ($request->isValid())
	{
		
		if (empty($ROW_VAR_NAME['ID_COLUMN_NAME']))
		{
			$ENTITY_NAME_dbtable->insert($ROW_VAR_NAME);
		}
		else
		{
			$ENTITY_NAME_dbtable->updateById($ROW_VAR_NAME);
		}
		$this->screenAlertQueued('success', 'The ENTITY_NICENAME_LOWERCASE was saved successfully.', date(DB_DATETIME_FORMAT, time() + 30), 'INDEX_MCA');
		$this->_redirect('INDEX_URL');
	}
	else
	{
		foreach ($request->getValidationErrors() as $error)
		{
			$this->screenAlert('error', $error);
		}
	}
}
$this->view->ENTITY_NAME = $ROW_VAR_NAME;
